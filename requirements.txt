-- ============================================
-- Migration Script for Request Types and Roles
-- ============================================
-- این فایل شامل کوئری‌های SQL برای اضافه کردن انواع درخواست جدید است
-- تاریخ: برای اضافه کردن TAPE_BACKUP و FLASH_DISK_BACKUP
-- ============================================
-- 
-- توجه مهم: در PostgreSQL نمی‌توان مقادیر جدید را در یک transaction به enum اضافه کرد
-- بنابراین باید هر ALTER TYPE را جداگانه و خارج از transaction اجرا کنید
-- ============================================

-- ============================================
-- روش 1: اگر enum request_type از قبل وجود دارد
-- ============================================
-- این کوئری‌ها را یکی یکی و جداگانه اجرا کنید (نه در یک transaction)
-- توجه: اگر مقدار از قبل وجود دارد، خطا می‌دهد که می‌توانید نادیده بگیرید

-- کوئری 1: اضافه کردن TAPE (اگر TAPE_BACKUP از قبل وجود دارد، می‌توانید آن را حذف کنید)
-- اگر خطای "already exists" دریافت کردید، نادیده بگیرید
ALTER TYPE request_type ADD VALUE 'TAPE';

-- کوئری 2: اضافه کردن USB_PORT (اگر FLASH_DISK_BACKUP از قبل وجود دارد، می‌توانید آن را حذف کنید)
-- اگر خطای "already exists" دریافت کردید، نادیده بگیرید
ALTER TYPE request_type ADD VALUE 'USB_PORT';

-- کوئری 3: اضافه کردن APP_INSTALL
-- اگر خطای "already exists" دریافت کردید، نادیده بگیرید
ALTER TYPE request_type ADD VALUE 'APP_INSTALL';

-- کوئری 4: اضافه کردن VDI_OPEN (اگر وجود ندارد)
-- اگر خطای "already exists" دریافت کردید، نادیده بگیرید
ALTER TYPE request_type ADD VALUE 'VDI_OPEN';

-- ============================================
-- روش 2: اگر enum request_type وجود ندارد
-- ============================================
-- اگر خطای "type request_type does not exist" دریافت کردید،
-- ابتدا enum را ایجاد کنید:

-- CREATE TYPE request_type AS ENUM (
--     'FILE_TRANSFER',
--     'BACKUP',
--     'VDI_OPEN',
--     'TAPE',
--     'USB_PORT',
--     'APP_INSTALL'
-- );

-- سپس اگر ستون request_type در جدول requests وجود ندارد، آن را اضافه کنید:
-- ALTER TABLE requests 
-- ADD COLUMN IF NOT EXISTS request_type request_type DEFAULT 'FILE_TRANSFER';

-- اگر ستون request_type از نوع VARCHAR است، آن را به enum تبدیل کنید:
-- ALTER TABLE requests 
-- ALTER COLUMN request_type TYPE request_type 
-- USING request_type::request_type;

-- ============================================
-- راهنمای اجرا:
-- ============================================
-- 1. به دیتابیس PostgreSQL متصل شوید:
--    psql -U postgres -d postgres
--
-- 2. هر کوئری را جداگانه اجرا کنید (نه همه در یک transaction)
--
-- 3. یا در pgAdmin:
--    Tools > Query Tool > هر کوئری را جداگانه اجرا کنید
--
-- 4. یا از طریق فایل:
--    psql -U postgres -d postgres -f requirements.txt
--    (اما توجه: ممکن است نیاز باشد هر ALTER TYPE را جداگانه اجرا کنید)
-- ============================================

-- ============================================
-- اضافه کردن Role جدید NETWORK_USB_ADMIN
-- ============================================
-- این کوئری را برای اضافه کردن Role جدید به enum user_role اجرا کنید

-- اضافه کردن NETWORK_USB_ADMIN به enum user_role
ALTER TYPE user_role ADD VALUE 'NETWORK_USB_ADMIN';

-- درخواست ویدئو کنفرانس (در صورت «already exists» نادیده بگیرید)
ALTER TYPE request_type ADD VALUE 'VIDEO_CONFRENCE';

-- نقش‌های ویدئو کنفرانس
ALTER TYPE user_role ADD VALUE 'V_REQUESTER';
ALTER TYPE user_role ADD VALUE 'VC_ACCEPTER';

-- درخواست ریستارت سرور (در صورت «already exists» نادیده بگیرید)
ALTER TYPE request_type ADD VALUE 'SERVER_RESTART';

-- ============================================
-- تغییرات دیتابیس برای ماژول شناسنامه سرورها
-- تاریخ اعمال: بعد از پیاده‌سازی ماژول مدیریت سامانه‌ها
-- ============================================

-- ── ۱. اضافه کردن ستون system_name به جدول contractor ──────────────────────
-- نام سامانه متعلق به پیمانکار است (نه سرور)
ALTER TABLE contractor
ADD COLUMN IF NOT EXISTS system_name VARCHAR(255);

-- ── ۲. اضافه کردن ستون req_user_ids به جدول contractor ────────────────────
-- برای نگهداری آرایه‌ای از idهای کاربران مالک این سامانه
ALTER TABLE contractor
ADD COLUMN IF NOT EXISTS req_user_ids JSONB DEFAULT '[]'::JSONB;

-- اعمال محدودیت برای معتبر بودن JSON به صورت آرایه
ALTER TABLE contractor
ADD CONSTRAINT IF NOT EXISTS valid_req_user_ids_contractor
CHECK (jsonb_typeof(req_user_ids) = 'array');

-- ── ۳. تغییر ساختار جدول backup_resources ──────────────────────────────────
-- حذف ستون‌های قدیمی که دیگر استفاده نمی‌شوند:
--   contractor_ids  (آرایه JSONB — جایگزین شده با contractor_id)
--   req_user_ids    (آرایه JSONB — به جدول contractor منتقل شد)
--   system_name     (به جدول contractor منتقل شد)

-- اضافه کردن کلید خارجی contractor_id
ALTER TABLE backup_resources
ADD COLUMN IF NOT EXISTS contractor_id INTEGER REFERENCES contractor(id) ON DELETE SET NULL;

-- حذف ستون‌های قدیمی (پس از مطمئن شدن از migration داده‌ها)
-- ALTER TABLE backup_resources DROP COLUMN IF EXISTS contractor_ids;
-- ALTER TABLE backup_resources DROP COLUMN IF EXISTS req_user_ids;
-- ALTER TABLE backup_resources DROP COLUMN IF EXISTS system_name;

-- ── ۴. ساختار نهایی جداول ──────────────────────────────────────────────────
-- جدول contractor (سامانه + پیمانکار):
--   id            SERIAL PRIMARY KEY
--   system_name   VARCHAR(255)          -- نام سامانه (اجباری در UI)
--   cont_name     VARCHAR(255)          -- نام پیمانکار
--   rep_name      VARCHAR(255)          -- نام نماینده
--   phone         VARCHAR(12)           -- شماره تماس
--   req_user_ids  JSONB DEFAULT '[]'    -- آرایه idهای کاربران مالک

-- جدول backup_resources (سرورها):
--   id               SERIAL PRIMARY KEY
--   ip               INET NOT NULL       -- آدرس IP (اجباری)
--   vmname           VARCHAR(255) NOT NULL
--   url              VARCHAR(255)
--   type             VARCHAR(50)
--   backup_operator  VARCHAR(255)
--   backup_period    VARCHAR(255)
--   contractor_id    INTEGER REFERENCES contractor(id) ON DELETE SET NULL

-- ── ۵. ایندکس پیشنهادی برای بهبود کارایی ──────────────────────────────────
CREATE INDEX IF NOT EXISTS idx_backup_resources_contractor_id
ON backup_resources(contractor_id);

-- ── ۶. نکته مهم درباره migration داده‌های قدیمی ───────────────────────────
-- اگر داده‌های قدیمی در backup_resources با contractor_ids (JSONB) دارید،
-- باید آن‌ها را به contractor_id (INTEGER) تبدیل کنید:
--
-- UPDATE backup_resources br
-- SET contractor_id = (
--   SELECT (jsonb_array_elements_text(br.contractor_ids))::integer
--   LIMIT 1
-- )
-- WHERE br.contractor_ids != '[]'::jsonb
--   AND br.contractor_id IS NULL;
-- ============================================

-- ============================================
-- تغییرات جدول contractor برای پشتیبانی از ۳ نماینده
-- ============================================

-- حذف ستون‌های قدیمی (rep_name و phone) و جایگزینی با ستون‌های شماره‌دار
-- اگر ستون‌های قدیمی وجود دارند:
ALTER TABLE contractor RENAME COLUMN rep_name TO rep_name1;
ALTER TABLE contractor RENAME COLUMN phone TO phone1;

-- اضافه کردن نماینده دوم و سوم
ALTER TABLE contractor ADD COLUMN IF NOT EXISTS rep_name2 VARCHAR(255);
ALTER TABLE contractor ADD COLUMN IF NOT EXISTS phone2 VARCHAR(12);
ALTER TABLE contractor ADD COLUMN IF NOT EXISTS rep_name3 VARCHAR(255);
ALTER TABLE contractor ADD COLUMN IF NOT EXISTS phone3 VARCHAR(12);

-- ============================================
-- هیچ تغییر دیتابیسی نیاز نیست.
-- منطق دسترسی در سرور پیاده‌سازی شده:
-- - کاربر با group_ids={0}: دسترسی کامل به همه سامانه‌ها
-- - کاربر همگروه با ثبت‌کننده: می‌تواند ببیند
-- - فقط ثبت‌کننده یا کاربر با group_ids={0}: می‌تواند ویرایش/حذف کند
-- ============================================

-- ============================================================
-- مجموعه کامل تغییرات دیتابیس برای ماژول شناسنامه سرورها

-- ============================================================

-- ────────────────────────────────────────────────────────────
-- مرحله ۱: تغییرات جدول contractor
-- ────────────────────────────────────────────────────────────

-- ۱-الف) اضافه کردن ستون نام سامانه
ALTER TABLE contractor
  ADD COLUMN IF NOT EXISTS system_name VARCHAR(255);

-- ۱-ب) اضافه کردن ستون req_user_ids (آرایه idهای کاربران مالک)
ALTER TABLE contractor
  ADD COLUMN IF NOT EXISTS req_user_ids JSONB DEFAULT '[]'::JSONB;

-- ۱-ج) محدودیت معتبر بودن JSON به صورت آرایه
DO $$
BEGIN
  IF NOT EXISTS (
    SELECT 1 FROM pg_constraint
    WHERE conname = 'valid_req_user_ids_contractor'
  ) THEN
    ALTER TABLE contractor
      ADD CONSTRAINT valid_req_user_ids_contractor
      CHECK (jsonb_typeof(req_user_ids) = 'array');
  END IF;
END $$;

-- ۱-د) تبدیل ستون‌های نماینده به فرمت شماره‌دار (rep_name1, phone1, ...)
--
-- اگر جدول contractor از قبل ستون‌های rep_name و phone دارد:
ALTER TABLE contractor RENAME COLUMN rep_name TO rep_name1;
ALTER TABLE contractor RENAME COLUMN phone    TO phone1;
--
-- اگر ستون‌های rep_name و phone وجود ندارند (جدول تازه):
-- ALTER TABLE contractor ADD COLUMN IF NOT EXISTS rep_name1 VARCHAR(255);
-- ALTER TABLE contractor ADD COLUMN IF NOT EXISTS phone1    VARCHAR(12);

-- ۱-ه) اضافه کردن نماینده دوم و سوم
ALTER TABLE contractor ADD COLUMN IF NOT EXISTS rep_name2 VARCHAR(255);
ALTER TABLE contractor ADD COLUMN IF NOT EXISTS phone2    VARCHAR(12);
ALTER TABLE contractor ADD COLUMN IF NOT EXISTS rep_name3 VARCHAR(255);
ALTER TABLE contractor ADD COLUMN IF NOT EXISTS phone3    VARCHAR(12);

-- ────────────────────────────────────────────────────────────
-- مرحله ۲: تغییرات جدول backup_resources
-- ────────────────────────────────────────────────────────────

-- ۲-الف) اضافه کردن کلید خارجی contractor_id
ALTER TABLE backup_resources
  ADD COLUMN IF NOT EXISTS contractor_id INTEGER
  REFERENCES contractor(id) ON DELETE SET NULL;

-- ۲-ب) ایندکس برای بهبود کارایی JOIN
